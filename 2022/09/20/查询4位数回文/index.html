查询4位数回文

itertools 强大的python 迭代器标准库

#获取26个字母的所有四位数回文 .eg:ababa
import itertools
#itertools 是一个python迭代器标准库

list = itertools.permutations("abcdefghijklmnopqrstuvwxyz",2)
words = []
for item in list: #item是列表，需要循环读成字符串
  word = ""
  
  for letter in item:
    word = word + letter
    #已经获取到前两位字母排列，只需要拼接后两位内容
  word = word + word[0] + word[1]+ word[0]
  print(word)
  words.append(word)
